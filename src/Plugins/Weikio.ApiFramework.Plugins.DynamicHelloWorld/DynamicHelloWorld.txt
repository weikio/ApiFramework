namespace FunctionFramework.Plugins.DynamicHelloWorld
{
    public class DynamicHelloWorldApi
    {
        public string SayHello()
        {
            return "Dynamic Hello from Api Framework!";
        }

        public DynamicReturnMessage CreateMessage(string name)
        {
            return new DynamicReturnMessage($"Hello {name}");
        }
    }

    public class DynamicReturnMessage
    {
        public DynamicReturnMessage(string result)
        {
            Result = result;
        }

        public string Result { get; set; }
    }
}
